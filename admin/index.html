Firestore Security Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAuth() { return request.auth != null; }
    function isOwner(uid) { return request.auth.uid == uid; }
    function isAdmin() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Users
    match /users/{uid} {
      allow read: if isAuth();
      allow create: if isOwner(uid);
      allow update: if isOwner(uid) || isAdmin();
      allow delete: if isAdmin();

      match /achievements/{aid} {
        allow read: if isAuth();
        allow write: if isOwner(uid) || isAdmin();
      }
      match /activity/{aid} {
        allow read: if isOwner(uid);
        allow write: if isOwner(uid);
      }
    }

    // Quizzes
    match /quizSets/{quizId} {
      allow read: if resource.data.visibility == 'public' || isAuth();
      allow create: if isAuth();
      allow update, delete: if isAuth() &&
        (resource.data.creatorId == request.auth.uid || isAdmin());

      match /questions/{qid} {
        allow read: if isAuth();
        allow write: if isAuth() &&
          get(/databases/$(database)/documents/quizSets/$(quizId)).data.creatorId
            == request.auth.uid;
      }
    }

    // Games
    match /games/{gameId} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update: if isAuth() &&
        (resource.data.hostId == request.auth.uid || isAdmin());

      match /players/{playerId} {
        allow read: if isAuth();
        allow create: if isAuth();
        // Players can only update their own answered/score fields
        allow update: if request.auth.uid == playerId ||
          get(/databases/$(database)/documents/games/$(gameId)).data.hostId
            == request.auth.uid;
      }
    }
  }
}
